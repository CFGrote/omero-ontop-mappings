  ?key a rdf:Property .

    }

    ?img a ome_core:Image;
         ome_core:annotation ?annotation .
    ?annotation ome_core:mapAnnotationValue ?map .
    ?map ome_core:key ?keystr;
         ome_core:value ?val .

    }
  limit 10
